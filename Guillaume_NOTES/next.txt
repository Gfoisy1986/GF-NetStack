Et tu as raison: **BearSSL n’a plus de base64.c séparé**, mais *il expose maintenant le codec Base64 dans `br_base64_*` via `codec/dec.c` et `codec/enc.c`*.  
Donc on peut utiliser **BearSSL pour encoder en Base64** sans ajouter de fichier externe.

# ✅ 1. Générer la clé Sec-WebSocket-Key (16 bytes → Base64)

```c
#include <bearssl.h>

static int ws_generate_key(char *out_b64, size_t out_sz)
{
    uint8_t rnd[16];
    br_hmac_drbg_context rng;

    /* Init DRBG with some entropy (use your TLS RNG or OS RNG) */
    uint8_t seed[32] = {0};
    br_hmac_drbg_init(&rng, &br_sha256_vtable, seed, sizeof(seed));

    br_hmac_drbg_generate(&rng, rnd, sizeof(rnd));

    /* Base64 encode using BearSSL */
    size_t len = br_base64_encode(out_b64, out_sz, rnd, sizeof(rnd));
    if (len == 0 || len >= out_sz)
        return -1;

    out_b64[len] = '\0';
    return 0;
}
```

✔ Utilise **BearSSL**  
✔ Pas besoin de `base64.c`  
✔ Pas besoin de dépendance externe

# ✅ 2. Calculer Sec-WebSocket-Accept

RFC6455 dit:

```
Sec-WebSocket-Accept = Base64( SHA1( key + GUID ) )
```

GUID fixe:

```
258EAFA5-E914-47DA-95CA-C5AB0DC85B11
```

Voici le code:

```c
static int ws_compute_accept(const char *key_b64, char *out_b64, size_t out_sz)
{
    static const char *GUID =
        "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";

    uint8_t sha1_out[20];
    br_sha1_context ctx;

    br_sha1_init(&ctx);
    br_sha1_update(&ctx, key_b64, strlen(key_b64));
    br_sha1_update(&ctx, GUID, strlen(GUID));
    br_sha1_out(&ctx, sha1_out);

    size_t len = br_base64_encode(out_b64, out_sz, sha1_out, sizeof(sha1_out));
    if (len == 0 || len >= out_sz)
        return -1;

    out_b64[len] = '\0';
    return 0;
}
```

✔ SHA‑1 BearSSL  
✔ Base64 BearSSL  
✔ Pas de dépendances externes
