Here we go — a clean, self‑contained **SQLite JSON wrapper** in C, with a `.h` and `.c` ready to drop into your SDK.

- Cross‑platform  
- Minimal API  
- JSON table output  
- Handle‑based (int → `sqlite3*`)  
- Caller frees JSON with `dbw_free_json()`  

You can later wire this into your TLS server and JSON protocol.

---

### `sqlite_wrapper.h`

```c
// sqlite_wrapper.h
// Minimal SQLite → JSON wrapper for cross-language SDKs

#ifndef SQLITE_WRAPPER_H
#define SQLITE_WRAPPER_H

#ifdef __cplusplus
extern "C" {
#endif

// Open a database. Returns handle >= 1 on success, <0 on error.
int dbw_open(const char *path);

// Close a database handle.
void dbw_close(int handle);

// Execute a statement that doesn't return rows (CREATE, INSERT, UPDATE, DELETE).
// Returns 0 on success, <0 on error.
int dbw_exec_ok(int handle, const char *sql);

// Execute a query and return a JSON table string.
// JSON format:
// {
//   "columns": ["col1","col2",...],
//   "rows": [
//     ["v11","v12",...],
//     ["v21","v22",...]
//   ]
// }
// On success: *json_out points to malloc'd string, return 0.
// Caller must free with dbw_free_json().
int dbw_exec_json(int handle, const char *sql, char **json_out);

// Execute a query that returns a single scalar value (first column of first row).
// Writes result as string into buf (null-terminated).
// Returns 0 on success, >0 if no rows, <0 on error.
int dbw_exec_scalar(int handle, const char *sql, char *buf, size_t bufsize);

// Free JSON string returned by dbw_exec_json().
void dbw_free_json(char *json);

#ifdef __cplusplus
}
#endif

#endif /* SQLITE_WRAPPER_H */
```

### `sqlite_wrapper.c`

```c
// sqlite_wrapper.c
// Minimal SQLite → JSON wrapper for cross-language SDKs

#include "sqlite_wrapper.h"

#include <sqlite3.h>
#include <stdlib.h>
#include <string.h>
#include <stdio.h>

#define DBW_MAX_HANDLES 64

typedef struct {
    int       in_use;
    sqlite3  *db;
} dbw_handle_t;

static dbw_handle_t g_dbw_handles[DBW_MAX_HANDLES];

// ---------------------------------------------------------
// Internal helpers
// ---------------------------------------------------------
static int dbw_alloc_handle(sqlite3 *db) {
    for (int i = 0; i < DBW_MAX_HANDLES; i++) {
        if (!g_dbw_handles[i].in_use) {
            g_dbw_handles[i].in_use = 1;
            g_dbw_handles[i].db     = db;
            return i + 1; // handle = index+1
        }
    }
    return -1;
}

static sqlite3 *dbw_get_db(int handle) {
    int idx = handle - 1;
    if (idx < 0 || idx >= DBW_MAX_HANDLES) return NULL;
    if (!g_dbw_handles[idx].in_use) return NULL;
    return g_dbw_handles[idx].db;
}

static void dbw_free_handle(int handle) {
    int idx = handle - 1;
    if (idx < 0 || idx >= DBW_MAX_HANDLES) return;
    g_dbw_handles[idx].in_use = 0;
    g_dbw_handles[idx].db     = NULL;
}

// Simple dynamic string builder
typedef struct {
    char  *buf;
    size_t used;
    size_t cap;
} dbw_str_t;

static int dbw_str_init(dbw_str_t *s) {
    s->cap  = 1024;
    s->used = 0;
    s->buf  = (char*)malloc(s->cap);
    if (!s->buf) return -1;
    s->buf[0] = '\0';
    return 0;
}

static int dbw_str_append(dbw_str_t *s, const char *text) {
    size_t len = strlen(text);
    if (s->used + len + 1 > s->cap) {
        size_t newcap = s->cap * 2;
        while (newcap < s->used + len + 1) newcap *= 2;
        char *nbuf = (char*)realloc(s->buf, newcap);
        if (!nbuf) return -1;
        s->buf = nbuf;
        s->cap = newcap;
    }
    memcpy(s->buf + s->used, text, len);
    s->used += len;
    s->buf[s->used] = '\0';
    return 0;
}

// NOTE: This is a minimal JSON string escaper (very basic).
static void dbw_json_escape(const char *in, dbw_str_t *out) {
    const char *p = in;
    char tmp[8];
    while (*p) {
        unsigned char c = (unsigned char)*p;
        switch (c) {
            case '\"': dbw_str_append(out, "\\\""); break;
            case '\\': dbw_str_append(out, "\\\\"); break;
            case '\b': dbw_str_append(out, "\\b");  break;
            case '\f': dbw_str_append(out, "\\f");  break;
            case '\n': dbw_str_append(out, "\\n");  break;
            case '\r': dbw_str_append(out, "\\r");  break;
            case '\t': dbw_str_append(out, "\\t");  break;
            default:
                if (c < 0x20) {
                    snprintf(tmp, sizeof(tmp), "\\u%04x", c);
                    dbw_str_append(out, tmp);
                } else {
                    char ch[2] = { (char)c, 0 };
                    dbw_str_append(out, ch);
                }
                break;
        }
        p++;
    }
}

// ---------------------------------------------------------
// Public API
// ---------------------------------------------------------
int dbw_open(const char *path) {
    sqlite3 *db = NULL;
    if (sqlite3_open(path, &db) != SQLITE_OK) {
        if (db) sqlite3_close(db);
        return -1;
    }
    int h = dbw_alloc_handle(db);
    if (h < 0) {
        sqlite3_close(db);
        return -2;
    }
    return h;
}

void dbw_close(int handle) {
    sqlite3 *db = dbw_get_db(handle);
    if (!db) return;
    sqlite3_close(db);
    dbw_free_handle(handle);
}

int dbw_exec_ok(int handle, const char *sql) {
    sqlite3 *db = dbw_get_db(handle);
    if (!db) return -1;

    char *errmsg = NULL;
    int rc = sqlite3_exec(db, sql, NULL, NULL, &errmsg);
    if (rc != SQLITE_OK) {
        if (errmsg) sqlite3_free(errmsg);
        return -2;
    }
    return 0;
}

typedef struct {
    dbw_str_t json;
    int       started_rows;
    int       col_count;
} dbw_json_ctx_t;

static int dbw_json_callback(void *udata, int argc, char **argv, char **colname) {
    dbw_json_ctx_t *ctx = (dbw_json_ctx_t*)udata;
    int i;

    if (ctx->col_count == 0) {
        ctx->col_count = argc;
        dbw_str_append(&ctx->json, "{\"columns\":[");
        for (i = 0; i < argc; i++) {
            if (i > 0) dbw_str_append(&ctx->json, ",");
            dbw_str_append(&ctx->json, "\"");
            if (colname[i]) dbw_json_escape(colname[i], &ctx->json);
            dbw_str_append(&ctx->json, "\"");
        }
        dbw_str_append(&ctx->json, "],\"rows\":[");
    }

    if (ctx->started_rows) {
        dbw_str_append(&ctx->json, ",");
    } else {
        ctx->started_rows = 1;
    }

    dbw_str_append(&ctx->json, "[");
    for (i = 0; i < argc; i++) {
        if (i > 0) dbw_str_append(&ctx->json, ",");
        dbw_str_append(&ctx->json, "\"");
        if (argv[i]) dbw_json_escape(argv[i], &ctx->json);
        dbw_str_append(&ctx->json, "\"");
    }
    dbw_str_append(&ctx->json, "]");

    return 0;
}

int dbw_exec_json(int handle, const char *sql, char **json_out) {
    sqlite3 *db = dbw_get_db(handle);
    if (!db || !json_out) return -1;

    dbw_json_ctx_t ctx;
    memset(&ctx, 0, sizeof(ctx));
    if (dbw_str_init(&ctx.json) < 0) return -2;

    char *errmsg = NULL;
    int rc = sqlite3_exec(db, sql, dbw_json_callback, &ctx, &errmsg);
    if (rc != SQLITE_OK) {
        if (errmsg) sqlite3_free(errmsg);
        free(ctx.json.buf);
        return -3;
    }

    if (ctx.col_count == 0) {
        // No rows, no columns
        dbw_str_append(&ctx.json, "{\"columns\":[],\"rows\":[]}");
    } else {
        dbw_str_append(&ctx.json, "]}");
    }

    *json_out = ctx.json.buf;
    return 0;
}

int dbw_exec_scalar(int handle, const char *sql, char *buf, size_t bufsize) {
    sqlite3 *db = dbw_get_db(handle);
    if (!db || !buf || bufsize == 0) return -1;

    sqlite3_stmt *stmt = NULL;
    int rc = sqlite3_prepare_v2(db, sql, -1, &stmt, NULL);
    if (rc != SQLITE_OK) {
        if (stmt) sqlite3_finalize(stmt);
        return -2;
    }

    rc = sqlite3_step(stmt);
    if (rc == SQLITE_ROW) {
        const unsigned char *txt = sqlite3_column_text(stmt, 0);
        if (!txt) {
            buf[0] = '\0';
        } else {
            strncpy(buf, (const char*)txt, bufsize - 1);
            buf[bufsize - 1] = '\0';
        }
        sqlite3_finalize(stmt);
        return 0;
    } else if (rc == SQLITE_DONE) {
        sqlite3_finalize(stmt);
        return 1; // no rows
    } else {
        sqlite3_finalize(stmt);
        return -3;
    }
}

void dbw_free_json(char *json) {
    if (json) free(json);
}
```
