# ✅ **Fortran95 FFI for `tls_v2.c`**

Save this as **`tls_v2.f90`**.

```fortran
!===============================================================
!  Fortran95 FFI wrapper for tls_v2.c
!===============================================================
MODULE tls_v2
  USE, INTRINSIC :: ISO_C_BINDING
  IMPLICIT NONE

  !-------------------------------------------------------------
  ! Callback interfaces (C function pointers)
  !-------------------------------------------------------------
  ABSTRACT INTERFACE
     SUBROUTINE cb_on_connect(id) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
     END SUBROUTINE cb_on_connect

     SUBROUTINE cb_on_disconnect(id) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
     END SUBROUTINE cb_on_disconnect

     SUBROUTINE cb_on_json(id, json, len) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
       TYPE(C_PTR), VALUE     :: json
       INTEGER(C_SIZE_T), VALUE :: len
     END SUBROUTINE cb_on_json
  END INTERFACE

  !-------------------------------------------------------------
  ! C struct: tlsv2_server_config_t
  !-------------------------------------------------------------
  TYPE, BIND(C) :: tlsv2_server_config_t
     INTEGER(C_INT) :: port
     TYPE(C_PTR)    :: cert_file
     TYPE(C_PTR)    :: key_file
     TYPE(C_FUNPTR) :: on_client_connected
     TYPE(C_FUNPTR) :: on_client_disconnected
     TYPE(C_FUNPTR) :: on_json_received
  END TYPE tlsv2_server_config_t

  !-------------------------------------------------------------
  ! Imported C functions
  !-------------------------------------------------------------
  INTERFACE
     FUNCTION tlsv2_server_run(cfg) BIND(C, NAME="tlsv2_server_run")
       USE ISO_C_BINDING
       TYPE(tlsv2_server_config_t), INTENT(IN) :: cfg
       INTEGER(C_INT) :: tlsv2_server_run
     END FUNCTION tlsv2_server_run

     SUBROUTINE tlsv2_send_json(id, json, len) BIND(C, NAME="tlsv2_send_json")
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
       TYPE(C_PTR), VALUE    :: json
       INTEGER(C_SIZE_T), VALUE :: len
     END SUBROUTINE tlsv2_send_json
  END INTERFACE

CONTAINS

  !-------------------------------------------------------------
  ! Helper: convert Fortran string → C string
  !-------------------------------------------------------------
  FUNCTION to_c_string(fstr) RESULT(cstr)
    CHARACTER(*), INTENT(IN) :: fstr
    TYPE(C_PTR) :: cstr
    CHARACTER(KIND=C_CHAR), DIMENSION(:), POINTER :: p
    INTEGER :: n

    n = LEN_TRIM(fstr)
    cstr = C_NULL_PTR

    CALL C_F_POINTER( C_LOC( fstr // CHAR(0) ), p, [n+1] )
    cstr = C_LOC(p(1))
  END FUNCTION to_c_string

END MODULE tls_v2
```

# ✅ **Fortran95 version of your C main()**

Save as **`server_main.f90`**:

```fortran
PROGRAM tls_server
  USE ISO_C_BINDING
  USE tls_v2
  IMPLICIT NONE

  TYPE(tlsv2_server_config_t) :: cfg

  INTERFACE
     SUBROUTINE on_connect(id) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
     END SUBROUTINE on_connect

     SUBROUTINE on_disconnect(id) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
     END SUBROUTINE on_disconnect

     SUBROUTINE on_json(id, json, len) BIND(C)
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: id
       TYPE(C_PTR), VALUE    :: json
       INTEGER(C_SIZE_T), VALUE :: len
     END SUBROUTINE on_json
  END INTERFACE

  !-------------------------------------------------------------
  ! Assign config
  !-------------------------------------------------------------
  cfg%port = 8443
  cfg%cert_file = to_c_string("server.pem")
  cfg%key_file  = to_c_string("server.key")

  cfg%on_client_connected    = C_FUNLOC(on_connect)
  cfg%on_client_disconnected = C_FUNLOC(on_disconnect)
  cfg%on_json_received       = C_FUNLOC(on_json)

  PRINT *, "Starting TLS server on port 8443..."
  CALL tlsv2_server_run(cfg)

END PROGRAM tls_server
```

# ✅ **Callback implementations**

```fortran
SUBROUTINE on_connect(id) BIND(C)
  USE ISO_C_BINDING
  INTEGER(C_INT), VALUE :: id
  PRINT *, "Client connected:", id
END SUBROUTINE on_connect

SUBROUTINE on_disconnect(id) BIND(C)
  USE ISO_C_BINDING
  INTEGER(C_INT), VALUE :: id
  PRINT *, "Client disconnected:", id
END SUBROUTINE on_disconnect

SUBROUTINE on_json(id, json, len) BIND(C)
  USE ISO_C_BINDING
  INTEGER(C_INT), VALUE :: id
  TYPE(C_PTR), VALUE    :: json
  INTEGER(C_SIZE_T), VALUE :: len

  CHARACTER(KIND=C_CHAR), POINTER :: buf(:)
  CHARACTER(LEN=:), ALLOCATABLE :: msg

  CALL C_F_POINTER(json, buf, [len])
  msg = TRANSFER(buf, msg, len)

  PRINT *, "From", id, ":", TRIM(msg)

  CALL tlsv2_send_json(id, json, len)
END SUBROUTINE on_json
```
