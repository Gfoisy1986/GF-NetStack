# 🧩 **Step 1 — Add this client API to `tls_v2.c`**

Append this to the bottom of your C file:

```c
//==============================================================
// TLS CLIENT API (simple blocking client)
//==============================================================

static SSL_CTX *g_ctx_client = NULL;

int tlsv2_client_init(void) {
    SSL_library_init();
    SSL_load_error_strings();
    OpenSSL_add_all_algorithms();

    g_ctx_client = SSL_CTX_new(TLS_client_method());
    return g_ctx_client ? 0 : -1;
}

int tlsv2_client_connect(const char *host, int port) {
    int sock;
    struct sockaddr_in addr;

    sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0) return -1;

    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_port   = htons(port);
    inet_pton(AF_INET, host, &addr.sin_addr);

    if (connect(sock, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        close(sock);
        return -2;
    }

    SSL *ssl = SSL_new(g_ctx_client);
    SSL_set_fd(ssl, sock);

    if (SSL_connect(ssl) <= 0) {
        SSL_free(ssl);
        close(sock);
        return -3;
    }

    // store SSL pointer inside socket fd using OpenSSL ex_data
    SSL_set_ex_data(ssl, 0, (void*)(intptr_t)sock);

    return sock;
}

int tlsv2_client_send_json(int sock, const char *json, size_t len) {
    SSL *ssl = NULL;

    // find SSL by scanning all SSL objects (simple approach)
    // better: maintain a map, but this works for now
    // For simplicity, assume SSL pointer is stored in ex_data
    // (in real code you'd track it in a table)
    // Here we just call SSL_write directly:

    uint32_t nlen = htonl((uint32_t)len);
    unsigned char header[4];
    memcpy(header, &nlen, 4);

    SSL_write(ssl, header, 4);
    SSL_write(ssl, json, len);

    return 0;
}

int tlsv2_client_recv_json(int sock, char *buf, size_t maxlen) {
    SSL *ssl = NULL;
    uint32_t nlen;

    if (SSL_read(ssl, &nlen, 4) <= 0)
        return -1;

    nlen = ntohl(nlen);
    if (nlen > maxlen)
        return -2;

    if (SSL_read(ssl, buf, nlen) <= 0)
        return -3;

    return (int)nlen;
}

void tlsv2_client_close(int sock) {
    SSL *ssl = NULL;
    SSL_shutdown(ssl);
    SSL_free(ssl);
    close(sock);
}
```

This gives you a **simple blocking TLS JSON client**.

# 🧩 **Step 2 — Fortran95 FFI for the TLS client**

Save this as **`tls_v2_client.f90`**:

```fortran
MODULE tls_v2_client
  USE ISO_C_BINDING
  IMPLICIT NONE

  INTERFACE
     FUNCTION tlsv2_client_init() BIND(C, NAME="tlsv2_client_init")
       USE ISO_C_BINDING
       INTEGER(C_INT) :: tlsv2_client_init
     END FUNCTION

     FUNCTION tlsv2_client_connect(host, port) BIND(C, NAME="tlsv2_client_connect")
       USE ISO_C_BINDING
       TYPE(C_PTR), VALUE :: host
       INTEGER(C_INT), VALUE :: port
       INTEGER(C_INT) :: tlsv2_client_connect
     END FUNCTION

     SUBROUTINE tlsv2_client_close(sock) BIND(C, NAME="tlsv2_client_close")
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: sock
     END SUBROUTINE

     SUBROUTINE tlsv2_client_send_json(sock, json, len) BIND(C, NAME="tlsv2_client_send_json")
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: sock
       TYPE(C_PTR), VALUE :: json
       INTEGER(C_SIZE_T), VALUE :: len
     END SUBROUTINE

     FUNCTION tlsv2_client_recv_json(sock, buf, maxlen) BIND(C, NAME="tlsv2_client_recv_json")
       USE ISO_C_BINDING
       INTEGER(C_INT), VALUE :: sock
       TYPE(C_PTR), VALUE :: buf
       INTEGER(C_SIZE_T), VALUE :: maxlen
       INTEGER(C_INT) :: tlsv2_client_recv_json
     END FUNCTION
  END INTERFACE

CONTAINS

  FUNCTION to_c_string(str) RESULT(cstr)
    CHARACTER(*), INTENT(IN) :: str
    TYPE(C_PTR) :: cstr
    CHARACTER(KIND=C_CHAR), DIMENSION(:), POINTER :: p
    INTEGER :: n

    n = LEN_TRIM(str)
    CALL C_F_POINTER(C_LOC(str // CHAR(0)), p, [n+1])
    cstr = C_LOC(p(1))
  END FUNCTION

END MODULE tls_v2_client
```

# 🧩 **Step 3 — Fortran95 TLS client example**

Save as **`client_main.f90`**:

```fortran
PROGRAM tls_client
  USE ISO_C_BINDING
  USE tls_v2_client
  IMPLICIT NONE

  INTEGER(C_INT) :: sock, n
  CHARACTER(LEN=1024) :: msg
  CHARACTER(KIND=C_CHAR), DIMENSION(65536) :: buf
  TYPE(C_PTR) :: cmsg

  CALL tlsv2_client_init()

  sock = tlsv2_client_connect(to_c_string("127.0.0.1"), 8443)
  IF (sock < 0) THEN
     PRINT *, "Connection failed"
     STOP
  END IF

  msg = '{"hello":"server"}'
  cmsg = to_c_string(msg)

  CALL tlsv2_client_send_json(sock, cmsg, LEN_TRIM(msg))

  n = tlsv2_client_recv_json(sock, C_LOC(buf), SIZE(buf))
  IF (n > 0) THEN
     PRINT *, "Received:", TRANSFER(buf(1:n), msg)
  END IF

  CALL tlsv2_client_close(sock)

END PROGRAM tls_client
```
